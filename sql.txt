select t.CDPRODUTO, t.VLPRECO, u.VLINDICEFINANCEIRO, u.NUCONVERSAOUNIDADE, c.VLINDICECLIENTE from TBUPITEMTABELAPRECO t
join TBUPPRODUTOUNIDADE u
on t.CDPRODUTO = u.CDPRODUTO
join TBUPCLIENTEEMPRESA c
on t.cdempresa = c.CDEMPRESA
where t.CDPRODUTO = 104312 and t.CDEMPRESA = 10 and t.FLATIVO = 'S' 
and u.FLATIVO = 'S' and u.VLINDICEFINANCEIRO is not null and c.CDCLIENTE = '10-1000181'




select t.CDPRODUTO, t.VLPRECO, u.VLINDICEFINANCEIRO, u.NUCONVERSAOUNIDADE, u.CDUNIDADE,
c.VLINDICECLIENTE, c.VLINDICETABELAPRECO from TBUPITEMTABELAPRECO t
left join TBUPPRODUTOUNIDADE u on t.CDPRODUTO = u.CDPRODUTO and u.CDEMPRESA = t.CDEMPRESA
join TBUPCLIENTEEMPRESA c
on t.cdempresa = c.CDEMPRESA and t.CDTABELAPRECO = c.CDTABELAPRECO
where t.CDPRODUTO in (698386) and t.CDEMPRESA = 10 and t.FLATIVO = 'S'
and u.FLATIVO = 'S' 
and c.CDCLIENTE = '10-1000181' and c.FLATIVO = 'S'
